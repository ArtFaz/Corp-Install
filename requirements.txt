# PyInstaller - Para compilar em .exe
pyinstaller>=6.0.0

# =============================================================================
#   COMANDOS DE COMPILAÇÃO (.exe)
# =============================================================================
#
#   Compilação Básica:
#   -------------------
#   pyinstaller --onefile --uac-admin --name "Provisionador" main.py
#
#   Compilação com Ícone Personalizado:
#   ------------------------------------
#   pyinstaller --onefile --uac-admin --name "Provisionador" --icon=icon.ico main.py
#
#   Opções Explicadas:
#   -------------------
#     --onefile    : Gera um único arquivo .exe portátil
#     --uac-admin  : Solicita elevação de privilégios automaticamente
#     --name       : Define o nome do executável gerado
#     --icon       : Define um ícone personalizado (arquivo .ico)
#     --noconsole  : Oculta a janela do terminal (NÃO use neste projeto CLI)
#
